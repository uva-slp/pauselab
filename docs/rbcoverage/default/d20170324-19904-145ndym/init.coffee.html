      });
    };
 
    Init.format = function(desc, id) {
      if (desc.length > 50) {
        desc = desc.substring(0, 50) + "...";
      }
      return "\"" + desc + "\" more at: " + window.location.host + '/idea/' + id;
    };
 
    Init.twtrShare = function() {
      return $('.twtr-btns').click(function() {
        var desc, id;
        desc = $(this).data('desc');
        id = $(this).data('id');
        desc = encodeURIComponent(Init.format(desc, id));
        return window.open("https://twitter.com/intent/tweet?text=" + desc, 'name', 'height=300,width=500');
      });
    };
 
    Init.redirect = function(href) {
      return window.location = href;
    };
 
    Init.rowClicker = function() {
      return $('.row-link').each(function() {
        $(this).click(function() {
          return Init.redirect($(this).data('href'));
        });
        return $(this).mouseover(function() {
          return $(this).css('cursor', 'pointer');
        });
      });
    };
